// Title: Final Prices With a Special Discount in a Shop
            // Difficulty: Easy
            // Language: Python3
            // Link: https://leetcode.com/problems/final-prices-with-a-special-discount-in-a-shop/

class Solution:
    def finalPrices(self, prices: List[int]) -> List[int]:

        # for index, price in prices
        for i,item in enumerate(prices):

            # setting up inner loop to start at i+1
            start = i + 1
            for nextItem in prices[start:]:
                if nextItem <= item:
                    discount=item-nextItem
                    prices[i]=discount
                    # once item is found, break out of inner loop
                    break
                
        return prices
